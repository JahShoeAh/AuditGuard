# AuditGuard Database Schema

> **Purpose:** Persistent storage for audit reports with wallet address correlation

---

## Overview

The database stores audit reports generated by autonomous agents, linked to the wallet address that deployed the smart contract. This enables users to view all audits for contracts they own.

---

## Data Model

```typescript
// packages/sdk/db/report-db.ts
export interface AuditReport {
  id: string;                    // Unique identifier: "report:{jobId}"
  jobId: string;                 // On-chain auction job ID
  contractAddress: string;       // Smart contract address (Hedera or EVM)
  deployerAddress: string;       // Wallet address that deployed (Hedera 0.0.NNN or EVM 0x...)
  hederaAccountId: string;       // Hedera account ID if available (0.0.NNNN)
  chain: string;                 // "hedera" or "ethereum"
  contractType: ContractType;    // "lending", "dex", "staking", "bridge", "vault"
  
  // Report content
  mdContent: string;             // Full markdown report content (or reference)
  contentHash: string;           // SHA3-256 hash for integrity verification
  cid: string;                   // IPFS/0g content identifier
  mdFilePath: string;            // Local file path if stored on disk
  
  // Agent metadata
  agentAddresses: string[];      // Addresses of agents that performed audit
  agentCount: number;            // Number of participating agents
  findingCount: number;          // Total findings count
  findingsBySeverity: {
    critical: number;
    high: number;
    medium: number;
    low: number;
    info: number;
  };
  
  // Timeline
  timestamp: number;             // Unix timestamp of report generation
  jobIdCreated: number;          // Timestamp when job was created
  jobIdCompleted: number;        // Timestamp when job was completed
  
  // Marketplace
  listingId?: number;            // DataMarketplace listing ID (optional)
  price?: bigint;                // Price in GUARD tokens
  purchasedBy?: string[];        // Array of buyer addresses
  
  // Metadata
  tags: string[];                // Tags for filtering: ["reentrancy", "overflow", etc.]
  source: "hcs" | "contract" | "manual";  // How this record was created
}
```

## Storage Options

### Option 1: JSON File (MVP - Recommended)

**Pros:** Simple, zero dependencies, easy debugging  
**Cons:** No concurrent writes, scaling limited

**Structure:**
```
project-root/
└── data/
    └── reports.json          // Array of AuditReport objects
```

**Operations:**
```typescript
// Read
const reports = JSON.parse(fs.readFileSync('./data/reports.json', 'utf8'));

// Write
fs.writeFileSync('./data/reports.json', JSON.stringify(reports, null, 2));

// Find by deployer
const userReports = reports.filter(r => r.deployerAddress === walletAddress);
```

---

### Option 2: LevelDB (Production)

**Pros:** Fast, ACID-compliant, handles concurrent access  
**Cons:** Requires dependencies, more complex

**Structure:**
```
project-root/
└── data/
    └── reports.db/           // LevelDB database directory
```

**Operations:**
```typescript
const level = require('level');
const db = level('./data/reports.db');

// Put
await db.put(`report:${jobId}`, report);

// Get
const report = await db.get(`report:${jobId}`);

// Query by deployer (usingsecondary index)
const reports = [];
for await (const [key, value] of db.iterator()) {
  if (key.startsWith('report:') && value.deployerAddress === walletAddress) {
    reports.push(value);
  }
}
```

---

### Option 3: PostgreSQL (Enterprise)

**Pros:** SQL queries, relationships, transactions  
**Cons:** External service required

**Schema:**
```sql
CREATE TABLE audit_reports (
  id TEXT PRIMARY KEY,
  job_id TEXT NOT NULL,
  contract_address TEXT NOT NULL,
  deployer_address TEXT NOT NULL,
  hedera_account_id TEXT,
  chain TEXT NOT NULL,
  contract_type TEXT NOT NULL,
  md_content TEXT NOT NULL,
  content_hash TEXT NOT NULL,
  cid TEXT,
  md_file_path TEXT,
  agent_addresses TEXT[],           -- Array of addresses
  agent_count INTEGER NOT NULL,
  finding_count INTEGER NOT NULL,
  findings_by_severity JSONB,
  timestamp BIGINT NOT NULL,
  job_id_created BIGINT,
  job_id_completed BIGINT,
  listing_id BIGINT,
  price NUMERIC,
  purchased_by TEXT[],
  tags TEXT[],
  source TEXT NOT NULL,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Index for fast lookups
CREATE INDEX idx_deployer_address ON audit_reports(deployer_address);
CREATE INDEX idx_contract_address ON audit_reports(contract_address);
CREATE INDEX idx_job_id ON audit_reports(job_id);
```

## Access Control

```typescript
// Verification: Only deployer can view report
function canViewReport(report: AuditReport, userAddress: string): boolean {
  return report.deployerAddress.toLowerCase() === userAddress.toLowerCase();
}

// If marketplace purchase enabled:
function canViewReport(report: AuditReport, userAddress: string): boolean {
  const isDeployer = report.deployerAddress?.toLowerCase() === userAddress?.toLowerCase();
  const isPurchaser = report.purchasedBy?.some(addr => addr.toLowerCase() === userAddress.toLowerCase());
  return isDeployer || isPurchaser || report.price === 0;
}
```

## Database Initialization

```typescript
// packages/sdk/db/init.ts
import fs from 'fs';
import path from 'path';

export function initDatabase(): void {
  const dataDir = path.join(process.cwd(), 'data');
  
  if (!fs.existsSync(dataDir)) {
    fs.mkdirSync(dataDir, { recursive: true });
  }
  
  // Initialize empty reports array (JSON option)
  const reportsPath = path.join(dataDir, 'reports.json');
  if (!fs.existsSync(reportsPath)) {
    fs.writeFileSync(reportsPath, JSON.stringify([], null, 2));
  }
  
  // Initialize reports directory
  const reportsDir = path.join(dataDir, 'reports');
  if (!fs.existsSync(reportsDir)) {
    fs.mkdirSync(reportsDir, { recursive: true });
  }
}
```

## Migration Strategy

**From MemStore → Database:**

1. Start database-backed store in parallel with Zustand
2. Mirror writes to both
3. Verify database persistence
4. Switch reads to database
5. Remove MemStore

```typescript
// packages/sdk/db/migration.ts
export async function migrateFromMemStore(): Promise<number> {
  const { activeJobs, reportMetadata } = useStore.getState();
  let count = 0;
  
  for (const [jobId, meta] of Object.entries(reportMetadata)) {
    const job = activeJobs[jobId];
    if (job && meta) {
      const report: AuditReport = {
        id: `report:${jobId}`,
        jobId,
        contractAddress: job.contractAddress,
        deployerAddress: meta.deployer,
        hederaAccountId: null,
        chain: job.contractChain || 'hedera',
        contractType: job.contractType || 'unknown',
        mdContent: '',  // Will be populated later
        contentHash: meta.contentHash,
        cid: meta.cid,
        mdFilePath: '',
        agentAddresses: [],
        agentCount: meta.agentCount || 0,
        findingCount: meta.findingCount || 0,
        findingsBySeverity: { critical: 0, high: 0, medium: 0, low: 0, info: 0 },
        timestamp: Date.now(),
        jobIdCreated: 0,
        jobIdCompleted: 0,
        tags: [],
        source: 'migration'
      };
      await saveReport(report);
      count++;
    }
  }
  
  return count;
}
```

## Security Considerations

1. **File Path Sanitization:** Validate all paths to prevent path traversal
2. **Access Control:** Verify wallet address ownership before serving reports
3. **Content Hash Verification:** Compare uploaded content with stored hash
4. **Encryption:** Consider encrypting report content for sensitive audits
5. **Rate Limiting:** Limit API queries to prevent abuse

```typescript
// Track database operations
export const dbMetrics = {
  writes: 0,
  reads: 0,
  errors: 0,
  totalSizeBytes: 0,
  lastWrite: 0

function logOp(op: string, details?: any) {
  console.log(`[DB] ${op}: ${JSON.stringify(details || {})}`);
  dbMetrics[op.toLowerCase()]++;
  dbMetrics.lastWrite = Date.now();
}
```
